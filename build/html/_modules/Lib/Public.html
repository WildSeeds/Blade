[文档]def readonly(*kw):
    '''
    传入表格中多组输入框元素，判断元素属性是否为只读，如果是返回true，如果不为只读，返回false
    :param kw:表格中输入框元素
    :return:
    '''

    for i in kw:
        if not i.getattribute('readonly'):
            return False
    return True

            return False
        return True

[文档]def setvalue(inputform:form,dic:dict,begin:str,end:str):
    '''
    通过输入数据XML自动生成表单的赋值语句
    :param inputform: 脚本需要的输入值序列
    :param dic: 脚本需要的字典
    :param begin: 起点
    :param end: 结束点
    :return:
    '''
    flag = False
    for (key,value) in dic.items():
        #print(key, value)
        name = key.split('_')[0]
        if key==begin or flag :
            flag =True
            if value:
                if '=' in value:
                    tempdict = {}
                    temps = value.spit(',')
                    for temp in temps:
                        tempdict[temp.split('=')[0]]=temp.split('=')[1]
                        inputform.setvalue(name,tempdict)
                elif ',' in value:
                    inputform.setvalue(name, value.split(','))
                #print('inputform.setvalue({0},{1})'.format(key,value))
                else:
                    inputform.setvalue(name,value)
        if key == end:
            return
